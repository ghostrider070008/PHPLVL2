Задание 5.
class A {
    public function foo() {
        static $x = 0;
        echo ++$x;
    }
}
$a1 = new A();
$a2 = new A();
$a1->foo(); //1
$a2->foo(); //2
$a1->foo(); //3
$a2->foo(); //4
// т.к. переменная x статическая, соответственно она привязана только к классу A, и при вызывании метода foo() переменная x увеличивается на 1

class A {
    public function foo() {
        static $x = 0;
        echo ++$x;
    }
}
class B extends A {
}
$a1 = new A;
$b1 = new B;
$a1->foo();// 1 т.к. вызывается метод foo() класса A, а переменная x статическая, т.е. привязана только к классу A.
$b1->foo(); // 1 т.к. вызывается метод foo() класса B, а переменная x статическая, т.е. привязана только к классу B.
$a1->foo(); // 2 т.к. вызывается метод foo() класса A, а переменная x статическая, т.е. привязана только к классу A.
$b1->foo(); // 2 т.к. вызывается метод foo() класса B, а переменная x статическая, т.е. привязана только к классу B.
